def to_usmf status

	usmf = USMF.new
	user = User.new

	x = status.split('@')

	usmf.service = 'Random' 
     	usmf.id = x[0]
     	usmf.text = x[2]
     	user.name = x[1]

     	usmf.user = user

     	usmf.links = []
     	usmf.to_users = []
           
	usmf
	
end
