def to_hash

	res = { 
          "service" => service, 
    		  "id" => id,
    			"geo" => geo,
    			"application" => application,
    			"location" => location,
    			"date" => date,
    			"source" => source,
    			"text" => text,
    			"description" => description,
    			"keywords" => keywords,
    			"category" => category,
    			"duration" => duration,
    			"likes" => likes,
    			"dislikes" => dislikes,
    			"favorites" => favorites,
    			"comments" => comments,
    			"rates" => rates,
    			"rating" => rating,
    			"min_rating" => min_rating,
    			"max_rating" => max_rating
		    }
  begin
  
    unless :user == nil
      res["user"] = user.to_hash
    end
  
  rescue
  
    res["user"] = nil
    logger.error("A corrupt user was found")
  
  end

    l = []
  
    unless :links == nil

      links.each do |link|

        l << link.to_hash

      end
  
      res["links"] = l
  
    end

    tu = []
  
    unless :to_users == nil

      to_users.each do |us|

        tu << us.to_hash

      end
  
      res["to_users"] = tu
  
    end      

	res

end
